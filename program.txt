PROCEDURE swap(x, y) IS
VAR z
BEGIN
    z := x;
    x := y;
    y := z;
END

PROCEDURE add1(a) IS
VAR y
BEGIN
    y := 1;
    a := a + y;
END

PROCEDURE add2(x) IS
BEGIN
    add1(x);
    add1(x);
END

PROCEDURE makeOdd(x) IS
VAR y
BEGIN
    y := x%2;
    IF y = 1 THEN
        x := 10;
    ENDIF
    add1(x);
END

PROGRAM IS
VAR x
BEGIN
    READ x;
    makeOdd(x);
    WRITE x;
END
