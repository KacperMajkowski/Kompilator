PROGRAM IS
VAR n, max, i, prime, rem
BEGIN
    n := 2;
    READ max;
    REPEAT
        prime := 1;
        i := 2;
        WHILE i < n DO
             rem := n%i;
             IF rem = 0 THEN
                prime := 0;
             ENDIF
             i := i + 1;
        ENDWHILE

        IF prime = 1 THEN
            WRITE n;
        ENDIF

        prime := 1;
        n := n+1;
    UNTIL n >= max;
END
