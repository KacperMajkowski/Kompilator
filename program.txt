PROCEDURE swap(x, y) IS
VAR z
BEGIN
    z := x;
    x := y;
    y := z;
END

PROCEDURE add1(a) IS
VAR y
BEGIN
    y := 1;
    a := a + y;
END

PROCEDURE add2(x) IS
BEGIN
    add1(x);
    add1(x);
END

PROCEDURE makeOdd(x) IS
VAR y
BEGIN
    y := x%2;
    IF y = 1 THEN
        x := x + 1;
    ENDIF
    add1(x);
END

PROCEDURE sort(x, y, z) IS
BEGIN
    IF x > y THEN
        swap(x, y);
    ENDIF

    IF y > z THEN
        swap(y, z);
    ENDIF

    IF x > y THEN
        swap(x, y);
    ENDIF
END

PROGRAM IS
VAR x, y, z
BEGIN
    READ x;
    READ y;
    READ z;
    sort(x, y, z);
    write3(x, y, z);
END
