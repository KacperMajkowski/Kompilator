PROCEDURE swap(x, y) IS
VAR z
BEGIN
    z := x;
    x := y;
    y := z;
END

PROCEDURE add1(x) IS
VAR y
BEGIN
    y := 1;
    x := x + y;
END

PROCEDURE makeOdd(x) IS
VAR y
BEGIN
    y := x % 2;
    IF y = 0 THEN
        x := x + 1;
    ENDIF
END

PROGRAM IS
VAR x, y, z, w
BEGIN
    READ x;
    READ y;
    swap(x, y);
    WRITE x;
    WRITE y;
END
